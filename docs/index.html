public String grandFatherName(People people){
    if(people != null) {
        var father = people.getFather();
        if(father != null) {
            var grandFather = father.getFather();
            if(grandFather != null){
                var grandFatherName = grandFather.getName();
                if(grandFatherName != null) {
                    return grandFatherName;
                }
            }
        }
    }
    return "";
}
